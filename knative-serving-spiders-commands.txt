### show knative running
k get pod -n knative-serving


### create sunshine1 service
kn service create sunshine1 --image=whitneyl042/rainbows:v1


### describe sunshine1 service
kn service describe sunshine1


### show knative serving yaml
cat knative-serving.yaml


### get all sunshine1 Kubernetes resources
kubectl get all | grep sunshine1


### get all sunshine1 Kubernetes resources
kubectl get pods


### watch the pods come up!
watch kubectl get pods


### create a custom route for the spiders website
kn service update sunshine1 --image whitneyl042/rainbows:v2 --tag sunshine1-00002=spiders --traffic sunshine1-00001=100


### send 10% of traffic to newest revision
kn service update sunshine1 --image whitneyl042/rainbows:v3 --traffic sunshine1-00001=90 --traffic @latest=10


### describe sunshine1 service
kn service describe sunshine1


### watch sunshine1 service
watch kn service describe sunshine1


### gradually rollout 100% of traffic to newest revision
kn service update sunshine1 --traffic @latest=100 --annotation-service serving.knative.dev/rollout-duration="60s"


### watch sunshine1 service
watch kn service describe sunshine1


### show serving yaml
kn service describe sunshine1 -o yaml


### rollback to last safe version
kn service update sunshine1 --traffic sunshine1-00001=100
